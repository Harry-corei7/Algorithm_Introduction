`难度: 中等`

你可以选择使用单链表或者双链表，设计并实现自己的链表。

1) 单链表中的节点应该具备两个属性：val 和 next 。
- val 是当前节点的值
- next 是指向下一个节点的指针/引用。
2) 如果是双向链表，则还需要属性 prev 以指示链表中的上一个节点。假设链表中的所有节点下标从 0 开始。

实现 MyLinkedList 类：

MyLinkedList() 初始化 MyLinkedList 对象。
- int get(int index) 获取链表中下标为 index 的节点的值。如果下标无效，则返回 -1 。
- void addAtHead(int val) 将一个值为 val 的节点插入到链表中第一个元素之前。在插入完成后，新节点会成为链表的第一个节点。
- void addAtTail(int val) 将一个值为 val 的节点追加到链表中作为链表的最后一个元素。
- void addAtIndex(int index, int val) 将一个值为 val 的节点插入到链表中下标为 index 的节点之前。如果 index 等于链表的长度，那么该节点会被追加到链表的末尾。如果 index 比长度更大，该节点将 不会插入 到链表中。
- void deleteAtIndex(int index) 如果下标有效，则删除链表中下标为 index 的节点。

**注意精简代码**
`1) addAtHead = addAtIndex(0, val)`
`2) addAtHead = addAtIndex(size, val)`

`(1)单链表`

`java语法`
```java
class MyLinkedList {

    // 创建单链表节点类
    class ListNode {
        int val;
        ListNode next;
        ListNode(int val) {
            this.val = val;
        }
    }
    // 创建成员变量
    int size; // 记录链表节点个数(不包括虚拟节点)
    ListNode header; // 创建虚拟头节点

    public MyLinkedList() {
        this.size = 0;
        this.header = new ListNode(0);
    }
    
    public int get(int index) {
        // 如果index非法, 返回-1
        if(index < 0 || index > size-1){ // index的范围[0, size-1], get不得过界
            return -1;
        }
        ListNode current = header;
        // 注意虚拟头节点是0, 真实head是1, 要找的node的index是index+1
        for(int i = 0; i < index+1; i++) { 
            current = current.next;
        }
        return current.val;
    }
    
    public void addAtHead(int val) {
        ListNode newNode = new ListNode(val);
        newNode.next = header.next;
        header.next = newNode;
        size++;
    }
    
    public void addAtTail(int val) {
        ListNode current = header;
        while(current.next != null){
            current = current.next;
        }
        ListNode newNode = new ListNode(val);
        current.next = newNode;
        size++;
    }
    
    public void addAtIndex(int index, int val) {
        // 判断index是否非法
        if(index < 0 || index > size){ // index的范围[0, size-1], add可以到size
            return;
        }
        ListNode current = header;

        for(int i = 0; i < index; i++){ // 到达输入的index的前一个节点
            current = current.next;
        }
        ListNode newNode = new ListNode(val);
        newNode.next = current.next;
        current.next = newNode;
        size++;
    }
    
    public void deleteAtIndex(int index) {
        // 判断index是否非法
        if(index < 0 || index > size-1){ // index的范围[0, size-1], delete不得过界
            return;
        }
        ListNode current = header;

        for(int i = 0; i < index; i++){ // 到达输入的index的前一个节点
            current = current.next;
        }
        current.next = current.next.next;
        size--;
    }
}
```

`python语法`

```python
class MyLinkedList(object):

    # 首先，创建链表节点的类
    class ListNode():
        def __init__(self, val=0, next=None):
            self.val = val
            self.next = next
    
    # 其次，完成链表类的初始化，添加两个属性: size和虚拟头节点
    def __init__(self):
        self.size = 0
        self.dummy = ListNode(-1)
        

    def get(self, index):
        """
        :type index: int
        :rtype: int
        """
        # 如果index不合法，超过当前节点index[0, size-1]
        if(index < 0 or index >= self.size):
            return -1
        else:
            current = self.dummy
            for i in range(-1, index):
                current = current.next
            return current.val


    def addAtHead(self, val):
        """
        :type val: int
        :rtype: None
        """
        temp = self.dummy.next
        new_head = ListNode(val)
        self.dummy.next = new_head
        self.dummy.next.next = temp
        self.size += 1


    def addAtTail(self, val):
        """
        :type val: int
        :rtype: None
        """
        current = self.dummy
        for i in range(self.size): # 指针直接移动到最后一个节点
            current = current.next
        new_tail = ListNode(val)
        current.next = new_tail
        self.size += 1
        

    def addAtIndex(self, index, val):
        """
        :type index: int
        :type val: int
        :rtype: None
        """
         # 如果index不合法,超过当前节点index以及尾节点.next[0, size]
        if(index < 0 or index > self.size):
            return None
        else:
            current = self.dummy
            for i in range(index): # 指针移动到要添加index的上一个节点
                current = current.next
            newNode = ListNode(val)
            temp = current.next
            current.next = newNode
            current.next.next = temp
            self.size += 1
        

    def deleteAtIndex(self, index):
        """
        :type index: int
        :rtype: None
        """
        # 如果index不合法，超过当前节点index[0, size-1]
        if(index < 0 or index >= self.size):
            return None
        else:
            current = self.dummy
            for i in range(index): # 指针移动到要删除的节点的上一个节点
                current = current.next
            temp = current.next.next
            current.next = temp
            self.size -= 1
```

`(2)双链表(未精简版)`

**注意: 双链表的addAtIndex**
1) 从前往后添加node, 是到输入index的**前一个节点**
- newNode.next = current.next;
- newNode.prev = current;
- current.next.prev = newNode;
- current.next = newNode;
2) 从后往前添加node, 是到输入index的**那一个节点**(当然也可以到"后一个节点", 需要prev.prev)
- newNode.prev = current.prev;
- newNode.next = current;
- current.prev.next = newNode;
- current.prev = newNode;

```java
class MyLinkedList {

    // 创建双链表节点类
    class ListNode {
        int val;
        ListNode next, prev;
        ListNode(int val){
            this.val = val;
        }
    }

    // 创建成员变量
    int size; // 记录链表节点个数(不包括虚拟节点)
    ListNode header; // 创建虚拟头节点
    ListNode tailer; // 创建虚拟尾节点

    public MyLinkedList() {
        this.size = 0;
        this.header = new ListNode(-1);
        this.tailer = new ListNode(-1);
        // 双链表初始化(非常重要, 防止出现null.next的情况)
        this.header.next = tailer;
        this.tailer.prev = header;
    }
    
    public int get(int index) {
        if(index < 0 || index > size-1) { // index范围[0, size-1], get不能越界
            return -1;
        }
        ListNode current;
        
        if(index <= (size/2)){
            current = header;
            for(int i = 0; i < index+1; i++) { // get(index)的节点的实际index是index+1
                current = current.next;
            }
        } else {
            current = tailer;
            for(int i = size+1; i > index+1; i--) { // get(index)的节点的实际index是index+1
                current = current.prev;
            }
        }
        return current.val;
    }
    
    public void addAtHead(int val) {
        ListNode current = header;
        ListNode newNode = new ListNode(val);
        // 双链表处理一个节点需要调整4个指针
        newNode.next = current.next;
        newNode.prev = current;
        current.next.prev = newNode;
        current.next = newNode;
        size++;
    }
    
    public void addAtTail(int val) {
        ListNode current = tailer;
        ListNode newNode = new ListNode(val);
        // 双链表处理一个节点需要调整4个指针
        newNode.next = tailer;
        newNode.prev = tailer.prev;
        tailer.prev.next = newNode;
        tailer.prev = newNode;
        size++;
    }
    
    public void addAtIndex(int index, int val) {
        if(index < 0 || index > size) {// index范围[0, size-1], add可以到size
            return;
        }
        if(index <= (size/2)){
            ListNode current = header;
            // 到达输入index的前一个节点
            for(int i = 0; i < index; i++){
                current = current.next;
            }
            ListNode newNode = new ListNode(val);
            newNode.next = current.next;
            newNode.prev = current;
            current.next.prev = newNode;
            current.next = newNode;
        } 
        else {
            ListNode current = tailer;
            // 到达输入index的那一个节点
            for(int i = size+1; i > index+1; i--){
                current = current.prev;
            }
            ListNode newNode = new ListNode(val);
            newNode.prev = current.prev;
            newNode.next = current;
            current.prev.next = newNode;
            current.prev = newNode;
        }
        size++;
    }
    
    public void deleteAtIndex(int index) {
        if(index < 0 || index > size-1) {// index范围[0, size-1], delete不能越界
            return;
        }
        if(index <= (size/2)){
            ListNode current = header;
            // 到达输入index的前一个节点
            for(int i = 0; i < index; i++){
                current = current.next;
            }
            current.next.next.prev = current;
            current.next = current.next.next;
        }
        else {
            ListNode current = tailer;
            // 到达输入index的后一个节点
            for(int i = size+1; i > index+2; i--){
                current = current.prev;
            }
            current.prev.prev.next = current;
            current.prev = current.prev.prev;
        }
        size--;
    }
}
```
