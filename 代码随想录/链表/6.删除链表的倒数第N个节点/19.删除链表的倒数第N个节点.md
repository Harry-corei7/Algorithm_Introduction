`难度: 中等`

给你一个链表，删除链表的倒数第 `n` 个结点，并且返回链表的头结点。

![image](https://github.com/user-attachments/assets/ea9313f9-f224-4cc9-9027-96ade9e35362)

输入：head = [1,2,3,4,5], n = 2
输出：[1,2,3,5]
示例 2：

输入：head = [1], n = 1
输出：[]
示例 3：

输入：head = [1,2], n = 1
输出：[1]

---
`java语法`
```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode removeNthFromEnd(ListNode head, int n) {
        // 建立一个虚拟头节点, 以便删除节点
        ListNode header = new ListNode(-1);
        header.next = head;
        // 建立快慢针
        ListNode slowIndex = header;
        ListNode fastIndex = header;
        // 确保快慢针之间隔了n
        for(int i = 0; i < n; i++){
            fastIndex = fastIndex.next;
        }
        // 同时移动
        while(fastIndex.next != null){
            fastIndex = fastIndex.next;
            slowIndex = slowIndex.next;
        }
        if(slowIndex.next != null){
            ListNode temp = slowIndex.next.next;
            slowIndex.next = temp;
        }
        // 注意: 一定要返回header.next, 而不是head
        return header.next;
    }
}
```
---
`python语法`
```python
class Solution(object):
    def removeNthFromEnd(self, head, n):
        """
        :type head: Optional[ListNode]
        :type n: int
        :rtype: Optional[ListNode]
        """
        # 设置虚拟头节点，一边删除节点
        dumy_head = ListNode(-1)
        dumy_head.next = head
        # 设置current指针
        current = dumy_head
        # 统计链表上的节点数量
        size = 0
        while(current.next != None):
            current = current.next
            size += 1
        current = dumy_head
        for i in range(size-n): # 到达要删除的节点的前一个节点
            current = current.next
        current.next = current.next.next
        return dumy_head.next
```
