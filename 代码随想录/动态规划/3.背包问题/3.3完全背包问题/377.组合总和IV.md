```python
class Solution(object):
    def combinationSum4(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: int
        """
        dp = [0]*(target + 1)
        # 初始化第一列
        dp[0] = 1 # 没放东西，算一种方案
        # 递推
        for j in range(0, target + 1):
            for i in range(0, len(nums)):
                if(j >= nums[i]):
                    dp[j] = dp[j - nums[i]] + dp[j]
        return dp[-1]
```
