class Solution(object):
    def backtracking(self, n, k, startIndex, result, path):
        if(len(path) == k):
            result.append(path[:])
            return
        for i in range(startIndex, n+1):
            path.append(i)
            self.backtracking(n, k, i+1, result, path)
            path.pop()


    def combine(self, n, k):
        """
        :type n: int
        :type k: int
        :rtype: List[List[int]]
        """
        result = [] # 存放结果
        path = []
        self.backtracking(n, k, 1, result, path)
        return result
